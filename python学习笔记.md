## 类
```python
class Calculator:

    def __init__(self,name,price=20):    ##类似于构造函数  需要引入参数时使用 无参数无法构建类  可以使用默认参数
        self.name=name
        self.price=price
    def sum(self,x,y):
        print(self.name)  ##self指向自身
        print(x+y)
    def mult(self,x,y):
        print(x*y)

calculator=Calculator()    ##创建对象 
self.price
self.name
```

## 元组 和 列表[]
```python
a_arr1=(1,3,5,7,9)

#遍历
for i in a_arr1 :
    print(i)

    print(a_arr2[index])

a=[1,2,4,5,6,7]
a.appended(8)   #在末尾添加8
a.insert(3,4)  #在索引值为3的地方添加4

a.index(3)   #第一个3的索引
```
